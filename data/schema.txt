# Predicates
id: string @index(hash) .
label: string @index(hash) @count .
type: string @index(hash) .
relation_label: string @index(hash) .
from: uid @reverse .
to: uid @reverse .
totalNodes: int .
totalEdges: int .

# Types
type Node {
  id
  label
}

type Relation {
  id
  type
  relation_label
  from
  to
}

type GraphStats {
  totalNodes
  totalEdges
}
